The Core — how concurrency control works in ~20 lines
class PromisePool {
  constructor(concurrency = 5) {
    this.concurrency = concurrency;   // ← your limit
  }

  run(tasks) {
    return new Promise((resolve) => {
      const total = tasks.length;
      const results = new Array(total);
      let completed = 0;
      let nextIndex  = this.concurrency; // pointer into queue

      const runTask = (index) => {
        Promise.resolve(tasks[index]())   // tasks are FUNCTIONS (lazy)
          .then(v  => results[index] = { status: 'fulfilled', value: v })
          .catch(r => results[index] = { status: 'rejected',  reason: r })
          .finally(() => {
            completed++;
            // Safe without locks — JS is single-threaded ✓
            if (completed === total)   resolve(results);
            else if (nextIndex < total) runTask(nextIndex++);
          });
      };

      // Seed initial burst — start exactly `concurrency` tasks
      for (let i = 0; i < Math.min(this.concurrency, total); i++) {
        runTask(i);
      }
    });
  }
}
